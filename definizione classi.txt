classe punto:
  attributi:
    x,y;
  metodi:
	conversione da punto a Qpoint


Funzioni esterne;
	distanza punto-retta;
    	distanza tra due punti (distanza dall'origine);
	distanza retta-retta
	intersezioni tra rette (ritorna != 0 -> perpendicolari);
	retta passante per due punti e ritorna una retta;
	
classe retta:
  attributi:
    double a;
    double b;
    double c;
    (forma esplicita : equazione retta : y = ax + c (per esempio y = 3x + 5));
    (forma implicita : equazione retta : ax+by+c=0 (per esempio 4x + 6y + 3 = 0));

  metodi:
    distanza da una retta;
    da implicita ad esplicita
    da esplicita ad implicita; (forse);
